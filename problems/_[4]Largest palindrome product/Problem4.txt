bool isPalindrome(unsigned long int number)
{
    unsigned long int tmp=number, invNumber=0;

    while (tmp)
    {
        invNumber = invNumber*10 + (tmp % 10);
        tmp /= 10;
    }
    return (invNumber == number ? true : false);
}
